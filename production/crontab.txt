# RSS2TLG Production Cron Jobs
# 
# Автоматический сбор RSS новостей каждую минуту
# 
# Формат crontab:
# MIN HOUR DAY MONTH DAYOFWEEK COMMAND
# 
# * - каждое значение
# */N - каждые N минут/часов/дней
# N,M,K - конкретные значения
# N-M - диапазон значений

# RSS Ingest - каждую минуту
* * * * * /usr/bin/php /home/engine/project/production/rss_ingest.php >> /home/engine/project/logs/cron_rss_ingest.log 2>&1
